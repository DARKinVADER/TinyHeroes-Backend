# Prize Eligibility — Minimum Deeds Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Let admins set a per-period minimum deed count that children must reach to qualify for a weekly or monthly prize; ineligible children are dimmed on the podium and prize slots for them show "Unclaimed".

**Architecture:** Two nullable int columns on `Family` (backend) carry the thresholds; a new `PATCH /api/families/mine/prize-rules` endpoint persists them. The frontend evaluates eligibility client-side using the existing `family` signal already loaded by `FamilyService`, avoiding any new API calls in the podium components.

**Tech Stack:** .NET 9 / EF Core (InMemory for tests), Angular 21 standalone components, Tailwind CSS, ngx-translate, xUnit + FluentAssertions.

[HttpPatch("mine/prize-rules")]
public async Task<ActionResult<FamilyResponse>> UpdatePrizeRules(SetPrizeRulesRequest req)
{
    var userId = GetUserId();
    var member = await db.FamilyMembers.FirstOrDefaultAsync(m => m.UserId == userId);
    if (member is null) return BadRequest("User does not belong to a family.");
    if (member.Role != FamilyRole.Admin) return Forbid();

    var family = await db.Families.FindAsync(member.FamilyId);
    if (family is null) return NotFound();

    family.WeeklyMinDeeds = req.WeeklyMinDeeds;
    family.MonthlyMinDeeds = req.MonthlyMinDeeds;
    await db.SaveChangesAsync();

    return Ok(new FamilyResponse(family.Id, family.Name, family.WeekStartDay, family.WeeklyMinDeeds, family.MonthlyMinDeeds));
}
```

import { Injectable, signal, inject } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { tap } from 'rxjs';
import { environment } from '../../../environments/environment';
import { Family, SetPrizeRulesRequest } from '../models/family.model';

@Injectable({ providedIn: 'root' })
export class FamilyService {
  private http = inject(HttpClient);
  private _family = signal<Family | null>(null);
  readonly family = this._family.asReadonly();

  loadFamily() {
    this.http.get<Family>(`${environment.apiUrl}/families/mine`).subscribe({
      next: (family) => this._family.set(family),
      error: () => this._family.set(null)
    });
  }

  updateFamily(name: string, weekStartDay: number) {
    return this.http.patch<Family>(`${environment.apiUrl}/families/mine`, { name, weekStartDay })
      .pipe(tap(f => this._family.set(f)));
  }

  updatePrizeRules(req: SetPrizeRulesRequest) {
    return this.http.patch<Family>(`${environment.apiUrl}/families/mine/prize-rules`, req)
      .pipe(tap(f => this._family.set(f)));
  }

  removeMember(userId: string) {
    return this.http.delete(`${environment.apiUrl}/families/mine/members/${userId}`);
  }

  deleteFamily() {
    return this.http.delete(`${environment.apiUrl}/families/mine`);
  }
}
```

export class PrizeRulesComponent implements OnInit {
  protected router = inject(Router);
  private familyService = inject(FamilyService);
  private authService = inject(AuthService);

  weeklyMin = signal(0);
  monthlyMin = signal(0);
  saving = signal(false);

  private isAdmin = computed(() => {
    const family = this.familyService.family();
    const userId = this.authService.user()?.userId ?? '';
    return family?.members.some(m => m.userId === userId && m.role === 'Admin') ?? false;
  });

  constructor() {
    effect(() => {
      const f = this.familyService.family();
      if (f) {
        this.weeklyMin.set(f.weeklyMinDeeds ?? 0);
        this.monthlyMin.set(f.monthlyMinDeeds ?? 0);
      }
    });
  }

  ngOnInit() {
    this.familyService.loadFamily();
    if (!this.isAdmin()) {
      this.router.navigate(['/settings']);
    }
  }

  incrementWeekly() { this.weeklyMin.update(v => v + 1); }
  decrementWeekly() { this.weeklyMin.update(v => Math.max(0, v - 1)); }
  incrementMonthly() { this.monthlyMin.update(v => v + 1); }
  decrementMonthly() { this.monthlyMin.update(v => Math.max(0, v - 1)); }

  save() {
    this.saving.set(true);
    this.familyService.updatePrizeRules({
      weeklyMinDeeds: this.weeklyMin(),
      monthlyMinDeeds: this.monthlyMin()
    }).subscribe({
      next: () => {
        this.saving.set(false);
        this.router.navigate(['/settings']);
      },
      error: () => this.saving.set(false)
    });
  }
}

export class PodiumComponent implements OnInit {
  private deedService = inject(DeedService);
  private childService = inject(ChildService);
  private prizeService = inject(PrizeService);
  private familyService = inject(FamilyService);
  private ts = inject(TranslateService);

  weeklyMin = computed(() => this.familyService.family()?.weeklyMinDeeds ?? 0);

  private isEligible(deedCount: number): boolean {
    const min = this.weeklyMin();
    return !min || deedCount >= min;
  }

  allRankings = computed(() => {
    const children = this.childService.children();
    const stats = this.deedService.stats();
    const sorted = children
      .map(c => ({
        ...c,
        weeklyCount: stats.find(s => s.childId === c.id)?.weeklyCount ?? 0
      }))
      .filter(c => c.weeklyCount > 0)
      .sort((a, b) => b.weeklyCount - a.weeklyCount);

    let rank = 0, prev = -1;
    return sorted.map((c, i) => {
      if (c.weeklyCount !== prev) { rank = i + 1; prev = c.weeklyCount; }
      return { ...c, rank, eligible: this.isEligible(c.weeklyCount) };
    });
  });

  eligibleRankings = computed(() => this.allRankings().filter(e => e.eligible));

  place1 = computed(() => this.allRankings().filter(e => e.rank === 1));
  place2 = computed(() => this.allRankings().filter(e => e.rank === 2));
  place3 = computed(() => this.allRankings().filter(e => e.rank === 3));

  hasAnyDeeds = computed(() => this.allRankings().length > 0);

  weeklyPrizes = computed(() =>
    this.prizeService.assignments()
      .filter(a => a.scope === 'weekly')
      .sort((a, b) => (a.rank ?? 0) - (b.rank ?? 0))
  );

  hasWeeklyPrizes = computed(() => this.weeklyPrizes().length > 0);

  prizeSlots = computed(() => {
    return this.weeklyPrizes().map(prize => {
      const rankHolder = this.allRankings().find(e => e.rank === prize.rank);
      return {
        rank: prize.rank ?? 0,
        prize,
        eligible: rankHolder?.eligible ?? false
      };
    });
  });

  ineligibleBannerText = computed(() => {
    const min = this.weeklyMin();
    if (!min) return null;
    const ineligible = this.allRankings().filter(e => !e.eligible);
    if (ineligible.length === 0) return null;
    const names = ineligible.map(e => e.name).join(', ');
    const plural = ineligible.length > 1 ? '' : 's';
    return this.ts.instant('PODIUM.INELIGIBLE_BANNER', { names, plural, min });
  });

  weekLabel = computed(() => {
    const today = new Date();
    const dayOfWeek = today.getDay();
    const monday = new Date(today);
    monday.setDate(today.getDate() - ((dayOfWeek + 6) % 7));
    const sunday = new Date(monday);
    sunday.setDate(monday.getDate() + 6);
    const lang = this.ts.currentLang || 'en';
    const fmt = (d: Date) => d.toLocaleDateString(lang, { month: 'short', day: 'numeric' });
    return this.ts.instant('PODIUM.WEEK_OF', { start: fmt(monday), end: fmt(sunday) });
  });

  ngOnInit() {
    this.deedService.loadStats();
    this.prizeService.loadAssignments();
    this.familyService.loadFamily();
    if (this.childService.children().length === 0) {
      this.childService.loadChildren();
    }
  }
}

                    <p class="text-xs font-bold text-yellow-700 uppercase tracking-wide">{{ 'PRIZES.MONTHLY_TITLE' | translate }}</p>
                    <p class="text-sm font-bold text-brand-text mt-0.5">{{ prize.emoji }} {{ prize.label }}</p>
                  </div>
                </div>

            </div>

            <!-- RIGHT col: Full Ranking -->
            <div>
              <h3 class="text-sm font-medium text-brand-text mt-6 lg:mt-0 mb-3">{{ 'MONTHLY.RANKING' | translate }}</h3>
              <div class="bg-white rounded-xl border border-brand-border divide-y divide-brand-border">
                @for (entry of rankedList(); track entry.childId) {
                  <div class="flex items-center gap-3 p-3"
                    [class.opacity-40]="monthlyMin() > 0 && entry.deedCount < monthlyMin()">
                    <span class="text-lg w-6">{{ entry.rank === 1 ? '🥇' : entry.rank === 2 ? '🥈' : entry.rank === 3 ? '🥉' : '' }}</span>
                    <div class="w-8 h-8 rounded-full bg-brand-cream flex items-center justify-center text-lg">
                      {{ getAvatar(entry.childId) }}
                    </div>
                    <span class="flex-1 text-sm font-medium text-brand-text">{{ entry.childName }}</span>
                    <span class="text-sm font-bold text-brand-orange">{{ entry.deedCount }}</span>
                  </div>

              </div>
            </div>
          </div>
        }
      </div>

export class MonthlyChampionComponent implements OnInit {
  private summaryService = inject(SummaryService);
  private childService = inject(ChildService);
  private prizeService = inject(PrizeService);
  private familyService = inject(FamilyService);
  private ts = inject(TranslateService);

  private currentMonth = signal<CurrentMonthStats | null>(null);

  monthlyMin = computed(() => this.familyService.family()?.monthlyMinDeeds ?? 0);

  rankedList = computed(() => this.currentMonth()?.rankings ?? []);

  noDeeds = computed(() => this.rankedList().length === 0);

  eligibleChampions = computed(() => {
    const min = this.monthlyMin();
    return this.rankedList().filter(r => r.rank === 1 && (!min || r.deedCount >= min));
  });

  eligibleChampion = computed(() => this.eligibleChampions().length > 0);

  monthLabel = computed(() => {
    const data = this.currentMonth();
    if (!data) return '';
    const lang = this.ts.currentLang || 'en';
    return new Date(data.year, data.month - 1).toLocaleDateString(lang, { month: 'long', year: 'numeric' });
  });

  monthName = computed(() => {
    const data = this.currentMonth();
    if (!data) return '';
    const lang = this.ts.currentLang || 'en';
    return new Date(data.year, data.month - 1).toLocaleDateString(lang, { month: 'long' });
  });

  monthlyPrize = computed(() =>
    this.prizeService.assignments().find(a => a.scope === 'monthly' && a.rank === null) ?? null
  );

  getAvatar(childId: string): string {
    return this.childService.children().find(c => c.id === childId)?.avatarEmoji ?? '🦸';
  }

  ngOnInit() {
    this.familyService.loadFamily();
    if (this.childService.children().length === 0) {
      this.childService.loadChildren();
    }
    if (this.prizeService.assignments().length === 0) {
      this.prizeService.loadAssignments();
    }
    this.summaryService.getCurrentMonth().subscribe({
      next: (stats) => this.currentMonth.set(stats),
      error: () => this.currentMonth.set(null)
    });
  }
}
